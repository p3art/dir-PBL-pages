import pandas as pd
#
#     Nantz =  ('22.728369 N', '120.299489 E')
# 北回歸線太陽館('23.522840 N', '120.414124 E')
#
def get_data(ecl_start, ecl_end) :
    #
    df1 = pd.read_excel('./z1_Ecl_ra_dec.xlsx', usecols=[0,1,2])
    #
    x_list = []
    y_list = []
    for i in range(ecl_start, ecl_end) :   
        df2 = df1.loc[df1['rec_no'] == i]
        xs = -df2.iloc[0,1]*15  # ra_hours(時)*15, 改取負值, 表示 in-look-out
        ys =  df2.iloc[0,2]     # dec_degrees(度) 
        x_list.append(xs)
        y_list.append(ys)
    # end for
    #
    return x_list, y_list
# end def
